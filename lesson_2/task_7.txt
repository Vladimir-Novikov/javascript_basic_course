Значения null и undefined равны друг другу, но не чему бы то ни было еще. Это правило
прописано в спецификации языка. При явном преобразовании в число null принимает значение 0, а undefined — NaN .

alert(null == 0); // false
alert(null === 0); //false
alert(+null == 0); //true              здесь преобразовал null в число и видим, что вернулось значение true
alert(+null === 0); //true           и такой же результат при строгом сравнении
alert(null == undefined); //true      
alert(null === undefined); //false          а здесь при строгом сравнении получили false, хотя при обычном сравнении было true - значит JS привел типы в предыдущем сравнении
alert(+null == undefined); //false         при преобразовании в число также получаем false
